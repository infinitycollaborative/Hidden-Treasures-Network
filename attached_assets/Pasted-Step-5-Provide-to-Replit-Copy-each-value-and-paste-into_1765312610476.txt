1. Firestore Security Rules

Go to Firestore Database (left sidebar)
Click Rules tab
Replace default rules with secure rules from your phase prompts

Example starter rules:
javascriptrules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if user is authenticated
    function isSignedIn() {
      return request.auth != null;
    }
    
    // Helper function to check user role
    function hasRole(role) {
      return isSignedIn() && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == role;
    }
    
    // Users collection - users can only read/write their own data
    match /users/{userId} {
      allow read: if isSignedIn();
      allow write: if isSignedIn() && request.auth.uid == userId;
    }
    
    // Students collection
    match /students/{studentId} {
      allow read: if isSignedIn();
      allow write: if hasRole('student') && request.auth.uid == studentId
                   || hasRole('mentor')
                   || hasRole('teacher')
                   || hasRole('platform_admin');
    }
    
    // More rules from your phase prompts...
  }
}
2. Storage Security Rules

Go to Storage (left sidebar)
Click Rules tab
Set up rules for file uploads

javascriptrules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /uploads/{userId}/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
